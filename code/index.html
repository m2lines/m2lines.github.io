import xarray as xr
import fsspec
# Datasets are hosted on globus as zarr files
def get_dataset(
    path,
    base_url="https://g-402b74.00888.8540.data.globus.org"
    ):
    mapper = fsspec.get_mapper(f"{base_url}/{path}.zarr")
    return xr.open_zarr(mapper, consolidated=True)
